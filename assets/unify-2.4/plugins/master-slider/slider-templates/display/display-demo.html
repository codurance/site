You need to put the following scripts just before the < ⁄ body> 
	<script type="text/javascript">		
	
		var slider = new MasterSlider();
		slider.setup('masterslider' , {
			width:507,
			height:286,
			speed:20,
			preload:0,
			space:2,
			view:'flow'
		});
		slider.control('arrows');	
		slider.control('bullets',{autohide:false});
			
	</script>
