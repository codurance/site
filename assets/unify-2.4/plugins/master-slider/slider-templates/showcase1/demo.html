	<script type="text/javascript">		
	
		var slider = new MasterSlider();
		slider.setup('masterslider' , {
			width:1024 ,
			height:580,
			space:0,
			fillMode:'fit',
			speed:25,
			preload:'all',
			view:'flow',
			loop:true
		});
		
		slider.control('arrows');	
		slider.control('bullets');
	</script>
